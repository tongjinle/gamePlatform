###### REQUEST #####
1. 登录
route:"login"
req:{
	username:string,
	password:string
}
res:{
	flag:boolean,
	token:string
}


2. 获取节点所有用户
route:"userlist"
req:{
	token:string
	pathnodeName:string
}
res:{
	flag:boolean,
	userlist:string[]
}

3. 登出
route:"logout"
req:null
res:{
	flag:boolean,
	username:string
}

##### NOTIFY ######
1. 发送聊天信息
route:"chat"
req:{
	from:string,
	to?:string,
	pathnodeName:string,
	message:string
}

##### PUSH #####
1. 发送聊天信息
route:"onChat"
res:{
	from:string,
	to?:string,
	pathnodeName:string,
	message:string,
	timeStamp:number
}

2. 用户加入
route:"onUserJoin"
res:{
	pathnodeName:string,
	username:string
}

3. 用户离开
route:"onUserLeave"
res:{
	pathnodeName:string,
	username:string
}
